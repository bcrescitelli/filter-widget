  <script>

    const rolePills = document.querySelectorAll('#role-filter .pill');

    let activeRole = "";


    rolePills.forEach(pill => {

      pill.addEventListener('click', () => {

        activeRole = pill.classList.contains('active') ? "" : pill.getAttribute('data-value');

        rolePills.forEach(p => p.classList.remove('active'));

        if (activeRole) pill.classList.add('active');

        updateResults();

      });

    });


    const segmentPills = document.querySelectorAll('#segment-filter .pill');

    const regionPills = document.querySelectorAll('#region-filter .pill');

    const typePills = document.querySelectorAll('#resource-type-filter .pill');

    const resultsDiv = document.getElementById('results');


    const resources = [

      { name: "SMB Template", segment: "SMB", region: "NAMER", types: ["Templates"], link: "https://linkedin.com" },

      { name: "ENT Template", segment: "ENT", region: "EMEAL", types: ["Learnings"], link: "https://linkedin.com" },

      { name: "MM Template", segment: "MM", region: "APAC", types: ["Wins", "Templates"], link: "https://linkedin.com" },

      { name: "Sales Navigator One Pager", segment: "LENT", region: "NAMER", types: ["Learnings", "Wins"], link: "https://linkedin.com" }

{ name: "Email Templates", segment: "SMB, MM, ENT, LENT, GCP", region: "NAMER, EMEAL, APAC", types: ["Learnings", "Wins"], link: "https://linkedin.com" }

{ name: "Sales Navigator One Pager", segment: "LENT", region: "NAMER", types: ["Learnings", "Wins"], link: "https://linkedin.com" }

    ];


    let activeSegment = "";

    let activeRegion = "";

    let activeTypes = [];


    segmentPills.forEach(pill => {

      pill.addEventListener('click', () => {

        activeSegment = pill.classList.contains('active') ? "" : pill.getAttribute('data-value');

        segmentPills.forEach(p => p.classList.remove('active'));

        if (activeSegment) pill.classList.add('active');

        updateResults();

      });

    });


    regionPills.forEach(pill => {

      pill.addEventListener('click', () => {

        activeRegion = pill.classList.contains('active') ? "" : pill.getAttribute('data-value');

        regionPills.forEach(p => p.classList.remove('active'));

        if (activeRegion) pill.classList.add('active');

        updateResults();

      });

    });


    typePills.forEach(pill => {

      pill.addEventListener('click', () => {

        const type = pill.getAttribute('data-value');

        if (pill.classList.contains('active')) {

          activeTypes = activeTypes.filter(t => t !== type);

          pill.classList.remove('active');

        } else {

          activeTypes.push(type);

          pill.classList.add('active');

        }

        updateResults();

      });

    });


    function clearFilters() {

      activeSegment = "";

      activeRegion = "";

      activeTypes = [];

      activeRole = "";

      [...rolePills, ...segmentPills, ...regionPills, ...typePills].forEach(p => p.classList.remove('active'));

      updateResults();

    }


    function updateResults() {

      resultsDiv.innerHTML = "";

      const filtered = resources.filter(resource => {

        const matchesSegment = !activeSegment || resource.segment === activeSegment;

        const matchesRegion = !activeRegion || resource.region === activeRegion;

        const matchesTypes = activeTypes.length === 0 || activeTypes.some(t => resource.types.includes(t));

        const matchesRole = true; // Update this logic if you add role-based filtering to resources

        return matchesSegment && matchesRegion && matchesTypes && matchesRole;

      });


      if (filtered.length === 0) {

        resultsDiv.innerHTML = "<p>No resources found.</p>";

        return;

      }


      filtered.forEach(resource => {

        const btn = document.createElement('button');

        btn.className = 'resource-button';

        btn.innerHTML = `<a href="${resource.link}" target="_blank">${resource.name} <span>↗</span></a>`;

        resultsDiv.appendChild(btn);

      });

    }


    updateResults();

  </script>
